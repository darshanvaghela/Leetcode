        for(ll i=0;i<n;i++)

                if(x>mx)
                {
                        mx=x;
                        id1=i;
                }
                if(x<mn)

                        mn=x;

                }
        }
        cout<<min({max(id1+1,id2+1),max(n-id1,n-id2),min(n-id1,id1+1)+min(n-id2,id2+1)})<<endl;

}
