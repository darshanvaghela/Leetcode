void solve()
{
        ll n,l,r;
        cin>>n>>l>>r;
        vll a(n);
        for(auto &x:a)
                cin>>x;
        sort(a.begin(),a.end());
        ll ans=0;
        for(ll i=0;i<n;i++)
        {
                auto l1=lower_bound(a.begin(),a.end(),l-a[i]);
                ll l;
                // if(l1==a.end())
                l=l1-a.begin();
                auto r1=upper_bound(a.begin(),a.end(),r-a[i]);
                // ll r;
                // if(r1==a.end())
                ll r=r1-a.begin();
                // r=n-r;
                r--;
                ans+=(r-l+1);
                // cout<<l<<r<<" ";
                if(i>=l and i<=r)
                        ans--;
        }
        cout<<ans/2<<endl;
        return;
}

int main()
{
    FAST;
    ll q;
    q=1;
    cin>>q;
    while(q--)
        solve();
    return 0;
}
