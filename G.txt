ll x,y,a,b;

bool chk(ll mid)
{
        ll mn=mid*a;
        ll remx=x-a*mid;
        ll remy=y-a*mid;
        ll other= remx/b+remy/b;
        return mn<=min(x,y) and other>=mid;
}

void solve()
{
        cin>>x>>y>>a>>b;
        if(a>b)
                swap(a,b);
        ll l=0,r=1e9,mid;
        ll ans=0;
        if(a==b)
        {
                cout<<min(x,y)/a<<endl;
                return;
        }
        b-=a;
        while(l<=r)
        {
                mid=(l+r)/2;
                if(chk(mid))
                {
                        ans=mid;
                        l=mid+1;
                }
                else
                        r=mid-1;
        }
        cout<<ans<<endl;
        return;
}   
int main()
{
    FAST;
    ll q;
    q=1;
    cin>>q;
    while(q--)
        solve();
    return 0;
}
